class Solution {
public:
    int maxArea(vector<int>& height) {
        // int ans = 0;
        int n = height.size();
        int l = 0, r = n - 1;
        int ans = (n - 1) * min(height[l], height[r]);
        while(l < r){
            int h = min(height[r], height[l]);
            ans = max(ans, (r - l) * h);
            if(height[l] > height[r])
                h = height[r --];
            else
                h = height[l ++];

        }
        return ans;
    }
};

                continue;

            int target = -nums[i];
            for(int l = i + 1, r = nums.size() - 1; l < nums.size(); l ++){
